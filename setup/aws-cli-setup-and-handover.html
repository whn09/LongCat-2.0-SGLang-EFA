1. 安装 AWS CLI 与 SSM 插件

1.1 macOS

# 方式一: Homebrew（推荐）
brew install awscli
brew install --cask session-manager-plugin

# 方式二: 官方 pkg（无 brew 时）
curl -o AWSCLIV2.pkg "https://awscli.amazonaws.com/AWSCLIV2.pkg"
sudo installer -pkg AWSCLIV2.pkg -target /
# SSM 插件 (Apple Silicon; Intel 把 mac_arm64 换成 mac):
curl -o smp.pkg "https://s3.amazonaws.com/session-manager-downloads/plugin/latest/mac_arm64/session-manager-plugin.pkg"
sudo installer -pkg smp.pkg -target /
sudo ln -sf /usr/local/sessionmanagerplugin/bin/session-manager-plugin /usr/local/bin/

1.2 Linux (x86_64)

curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o awscliv2.zip
unzip -q awscliv2.zip && sudo ./aws/install
# SSM 插件 (Ubuntu/Debian):
curl -o smp.deb "https://s3.amazonaws.com/session-manager-downloads/plugin/latest/ubuntu_64bit/session-manager-plugin.deb"
sudo dpkg -i smp.deb
# (AL2023/RHEL 系: 把 ubuntu_64bit/...deb 换成 linux_64bit/session-manager-plugin.rpm, 用 dnf install)

1.3 Windows

下载并运行两个 MSI： https://awscli.amazonaws.com/AWSCLIV2.msi 与 https://s3.amazonaws.com/session-manager-downloads/plugin/latest/windows/SessionManagerPluginSetup.exe

1.4 验证

aws --version                       # aws-cli/2.x
session-manager-plugin --version    # 1.2.x

2. 创建研发组 IAM 用户与授权（SRE 操作）

🔑 权限方案：一个托管策略 = PowerUserAccess AWS 托管策略 PowerUserAccess（arn:aws:iam::aws:policy/PowerUserAccess） 就是"研发最大权限"的标准答案：除 IAM/Organizations 账号管理外的全部服务操作全开 （EC2/S3/SSM/ECR/Bedrock/CloudWatch…），完全覆盖研发的日常工作 （登节点/调试/查容量/读写模型桶）。集群开机（CloudFormation 建栈，涉及创建 IAM 角色） 保留在 SRE 职责内，研发无需额外 IAM 权限。

2.1 创建用户组与用户（控制台或 CLI 均可，以下为 CLI）

# 1. 建组并挂 PowerUserAccess
aws iam create-group --group-name h200-dev
aws iam attach-group-policy --group-name h200-dev \
  --policy-arn arn:aws:iam::aws:policy/PowerUserAccess

# 2. 建用户入组（每个研发一个, 不要共享账号）
aws iam create-user --user-name dev-zhangsan
aws iam add-user-to-group --group-name h200-dev --user-name dev-zhangsan

# 3a. CLI 访问密钥（只显示一次, 安全传递给本人）
aws iam create-access-key --user-name dev-zhangsan

# 3b. 控制台密码（可选, 首登强制改密）
aws iam create-login-profile --user-name dev-zhangsan \
  --password '<临时密码>' --password-reset-required
⚠️ 安全底线（SRE 交接前确认） ① 每人独立用户，建议不共享 AK；② 控制台登录建议开 MFA；③ AK 定期轮换； ④ 账单/成本查看如需要，另加 AWSBillingReadOnlyAccess。

3. 配置 CLI 凭证与验证（研发组）

aws configure
# AWS Access Key ID:     <发给你的 AK>
# AWS Secret Access Key: <发给你的 SK>
# Default region name:   us-west-2        ← 本项目主区域
# Default output format: json

# 验证身份与权限:
aws sts get-caller-identity                       # 应显示你的 user ARN
aws ec2 describe-instances --region us-west-2 \
  --query 'Reservations[].Instances[].InstanceId' # 能列出实例 = EC2 权限 OK
aws s3 ls                                          # 能列桶 = S3 权限 OK

多区域/多身份可用 profile：aws configure --profile use2， 使用时 aws ... --profile use2 或 export AWS_PROFILE=use2。

4. 研发组日常操作速查

4.1 登录 GPU 节点（私有子网，全部走 SSM，无需 VPN/跳板机/开 22 端口）

# 交互 shell:
aws ssm start-session --target i-xxxx --region us-west-2

# SSH/scp/VSCode Remote（需节点上已有你的公钥, 开机时 KeyName 参数注入）~/.ssh/config:
Host gpu-*
  User ec2-user
  ProxyCommand aws ssm start-session --target %h --document-name AWS-StartSSHSession --parameters portNumber=22
# 用法: ssh gpu-i-xxxx

# 端口转发调试推理服务（本地 curl localhost:8000 直达节点 router）:
aws ssm start-session --target i-xxxx --document-name AWS-StartPortForwardingSession \
  --parameters '{"portNumber":["8000"],"localPortNumber":["8000"]}'

# 批量对多节点执行命令:
aws ssm send-command --instance-ids i-a i-b --document-name AWS-RunShellScript \
  --parameters 'commands=["nvidia-smi -L"]' --region us-west-2

4.2 查看 CB 容量 / 模型权重

aws ec2 describe-capacity-reservations --region us-west-2 \
  --query 'CapacityReservations[?ReservationType==`capacity-block`].{id:CapacityReservationId,state:State,az:AvailabilityZone,n:TotalInstanceCount,end:EndDate}' --output table
aws s3 ls s3://<模型桶>/models/
